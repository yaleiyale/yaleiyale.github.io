    顺序表

    建于：2022-07-30 20:46:12 Saturday 1717字 6分
    CC BY 4.0（除特别声明或转载）

    定义

    用顺序存储的方式实现的线性表。

    顺序存储。把逻辑上相邻的元素存储在物理位置上也相邻的存储单元中，元素之间的关系由存储单元的邻接关系来体现。

    实现

    1. 静态分配

    静态存储无法扩展

    类型定义

    #define MaxSize 10 //定义最大长度
typedef struct
{
    int data[MaxSize]; //用静态的“数组”存放数据元素
    int length;        //顺序表的当前长度
} SeqList;              //顺序表的类型定义

    初始化

    void InitList(SqList &L)
{
    for (int i = 0; i < MaxSize; i++)
        L.data[i] = 0; //将所有数据元素设置为默认值
    L.length = 0;      //顺序表初始化长度为0
}

    2. 动态分配

    类型定义

    #define Initsize 10 //顺序表的初始长度
typedef struct
{
    int *data;   //指示动态分配数组的指针
    int maxSize; //顺序表的最大容量
    int length;  //顺序表的当前长度
} SeqList;       //顺序表的类型定义

    初始化

    void InitList(SeqList &L)
{
    //用ma110c函数申请一片连续的存储空间
    L.data = (int *)malloc(Initsize * sizeof(int));
    L.length = 0;
    L.maxSize = Initsize;
}

    增加动态数组的长度

    void IncreaseSize(SeqList& L, int len)
{
	int* p = L.data;
	int newLength = L.maxSize + len;
	L.data = (int*)malloc(newLength * sizeof(int));
	if (nullptr == L.data)
		return;
	for (int i = 0; i < L.length; i++)
		L.data[i] = p[i];        // 将数据复制到新区域
	L.maxSize = newLength; //顺序表最大长度增加1en
	free(p);                     //释放原来的内存空间
}

    通用的增删查操作

    插入操作 1

    bool ListInsert(SeqList &L, int i, int e)
{
    if (i < 1 || i > L.length + 1) //判断i的范围是否有效
        return false;
    if (L.length >= MaxSize) //存储空间满则不能插入
        return false;
    for (int j = L.length; j >= i; j--)
        L.data[j] = L.data[j - 1]; //将第i个元素及之后的元素后移
    L.data[i - 1] = e;             //在位置i放入e
    L.length++;
    return true;
}

    删除操作 2

    bool ListDelete(SeqList &L, int i, int &e)
{
    if (i < 1 || i > L.length) //判断i的范围是否有效
        return false;
    e = L.data[i - 1];                 //将被删除的元素赋值给e
    for (int j = i; j < L.length; j++) //将第1个位置后的元素前移
        L.data[j - 1] = L.data[j];
    L.length--; //线性表长度减1
    return true;
}

    按位查找 3

    int GetElem(SeqList L, int i)
{
    return L.data[i - 1];
}

    按值查找 4

    int LocateElem(SeqList L, int e)
{
    for (int i = 0; i < L.length; i++)
    {
        if (L.data[i] == e) //本课程中不纠结"=="运算符对于结构类型的可行性
            return i + 1;
    }
    return 0;
}
