    定义

    每个结点除了存放数据元素外，还要存储指向下一个节点的指针

    typedef struct LNode
{
	int data;
	LNode* next;
} LNode, * LinkedList; // LinkedList表示LNode型指针，等价于LNode*

    基本操作的实现

    带头节点（推荐使用）

    初始化

    bool InitList(LinkedList& L)
{
	L = (LNode*)malloc(sizeof(LNode));
	if (L == nullptr)
		return false;
	L->next = nullptr;
	return true;
}

    判空

    bool Empty(LinkedList L)
{
	return (L->next == nullptr);
}

    按位序插入 1

    bool ListInsert(LinkedList& L, int i, int e)
{
	if (i < 1)
		return false;
	LNode* p;
	int j = 0;
	p = L;
	while (p != nullptr && j < i - 1) //循环找到第i-1个节点
	{
		p = p->next;
		j++;
	}
	if (p == nullptr)
		return false;
	LNode* s = (LNode*)malloc(sizeof(LNode));
	if (nullptr == s)
		return false;
	s->data = e;
	s->next = p->next;
	p->next = s; //将s连接到p之后
	return true;
}

    按位序删除 2

    bool ListDelete(LinkedList& L, int i, int& e)
{
	if (i < 1)
		return false;
	LNode* p;
	int j = 0;
	p = L;
	while (p != nullptr && j < i - 1) //循环找到第i-1个节点
	{
		p = p->next;
		j++;
	}
	if (p == nullptr)
		return false;
	if (p->next == nullptr) //第i-1个结点之后已无其他结点
		return false;
	LNode* q = p->next;
	e = q->data;
	p->next = q->next;
	free(q);
	return true;
}

    按位查找

    LNode* GetElem(LinkedList L, int i)
{
	if (i < 0)
		return nullptr;
	LNode* p;
	int j = 0;
	p = L;
	while (p != nullptr && j < i)
	{
		p = p->next;
		j++;
	}
	return p;
}

    按值查找

    LNode* LocateElem(LinkedList L, int e)
{
	LNode* p = L->next;
	while (p != nullptr && p->data != e)
		p = p->next;
	return p;
}

    求表长

    int Length(LinkedList L)
{
	int len = 0;
	LNode* p = L;
	while (p->next != nullptr)
	{
		p = p->next;
		len++;
	}
	return len;
}

    尾插法建立单链表

    bool List_TailInsert(LinkedList& L)
{
	L = (LNode*)malloc(sizeof(LNode)); //建立头结点
	if (nullptr == L)
		return false;
	LNode* s, * r = L;                   // r为表尾指针
	int x;
	std::cin >> x;
	while (x != 9999)
	{
		s = (LNode*)malloc(sizeof(LNode));
		if (nullptr == s)
			return false;
		s->data = x;
		r->next = s;
		r = s; // r指向新的表尾结点
		std::cin >> x;
	}
	r->next = nullptr;
	return true;
}

    头插法建立单链表

    重要应用：链表的逆置

    bool List_HeadInsert(LinkedList& L)
{
	LNode* s;
	int x;
	L = (LNode*)malloc(sizeof(LNode)); //建立头结点
	if (nullptr == L)
		return false;
	L->next = nullptr;                  //初始化为空链表，不让头节点的指针域指向随机地址
	std::cin >> x;
	while (x != 9999)
	{
		s = (LNode*)malloc(sizeof(LNode));
		if (nullptr == s)
			return false;
		s->data = x;
		s->next = L->next;
		L->next = s;
		std::cin >> x;
	}
	return true;
}

    通用操作

    指定结点的后插操作 3

    bool InsertNextNode(LNode* p, int e)
{
	if (p == nullptr)
		return false;
	LNode* s = (LNode*)malloc(sizeof(LNode));
	if (s == nullptr) //内存分配失败
		return false;
	s->data = e;
	s->next = p->next;
	p->next = s;
	return false;
}

    指定结点的前插操作 4

    bool InsertPriorNode(LNode* p, int e)
{
	if (p == nullptr)
		return false;
	LNode* s = (LNode*)malloc(sizeof(LNode));
	if (s == nullptr)
		return false;
	s->data = e;
	s->next = p->next;
	p->next = s; //插入新结点
	s->data = p->data;
	p->data = e; //元素互换
	return true;
}

    指定结点的删除 5

    bool DeleteNode(LNode* p)
{
	if (p == nullptr)
		return false;
	LNode* q = p->next;
	p->data = p->next->data;
	p->next = q->next;
	free(q);
	return true;
}

    按位序插入 6

    bool ListInsert(LinkedList& L, int i, int e)
{
	if (i < 1)
		return false;
	if (i == 1) //插入第一个结点的操作与其他结点操作不同
	{
		LNode* s = (LNode*)malloc(sizeof(LNode));
		if (nullptr == s)
			return false;
		s->data = e;
		s->next = L;
		L = s; //头指针指向新结点
		return true;
	}
	LNode* p;
	int j = 1;
	p = L;
	while (p != nullptr && j < i - 1) //循环找到第i-1个节点
	{
		p = p->next;
		j++;
	}
	if (p == nullptr)
		return false;
	LNode* s = (LNode*)malloc(sizeof(LNode));
	if (nullptr == s)
		return false;
	s->data = e;
	s->next = p->next;
	p->next = s; //将s连接到p之后
	return true;
}
